# ========================================
# COMANDOS PARA INSTALAR EN OTRO PC
# Copia y pega estos comandos en orden
# ========================================

# ===== PASO 1: VERIFICAR PYTHON =====
python --version
# Si no tienes Python, descárgalo de: https://www.python.org/downloads/
# IMPORTANTE: Marca "Add Python to PATH" durante la instalación


# ===== PASO 2: IR A LA CARPETA DEL PROYECTO =====
cd "C:\ruta\a\tu\proyecto\Inperia"
# Cambia la ruta según donde tengas el proyecto


# ===== PASO 3: ACTUALIZAR PIP =====
python -m pip install --upgrade pip


# ===== PASO 4: INSTALAR DEPENDENCIAS PYTHON =====

# Opción A: Si tienes requirements.txt
pip install -r requirements.txt

# Opción B: Instalar manualmente (si no tienes requirements.txt)
pip install openai-whisper
pip install PyQt5
pip install pyaudio
pip install numpy
pip install sounddevice

# Si pyaudio da error en Windows, usa:
pip install pipwin
pipwin install pyaudio


# ===== PASO 5: INSTALAR FFMPEG =====

# Opción A: Con Chocolatey (MÁS FÁCIL)
# 1. Abre PowerShell COMO ADMINISTRADOR
# 2. Si no tienes Chocolatey, instálalo primero con este comando:
Set-ExecutionPolicy Bypass -Scope Process -Force; [System.Net.ServicePointManager]::SecurityProtocol = [System.Net.ServicePointManager]::SecurityProtocol -bor 3072; iex ((New-Object System.Net.WebClient).DownloadString('https://community.chocolatey.org/install.ps1'))

# 3. Luego instala ffmpeg:
choco install ffmpeg

# Opción B: Manual
# Descarga desde: https://github.com/BtbN/FFmpeg-Builds/releases
# Extrae a C:\ffmpeg\
# Añade C:\ffmpeg\bin al PATH del sistema
# Ver INSTALAR_FFMPEG.md para detalles


# ===== PASO 6: VERIFICAR FFMPEG =====
ffmpeg -version
# IMPORTANTE: Si da error, REINICIA el terminal y vuelve a intentar


# ===== PASO 7: PROBAR INSTALACIÓN Y DESCARGAR MODELO =====
python test_whisper_instalacion.py
# ESTO DESCARGARÁ EL MODELO WHISPER MEDIUM (~1.5GB)
# Puede tardar 5-15 minutos la primera vez


# ===== PASO 8: EJECUTAR LA APLICACIÓN =====
python main.py
# O según tu archivo principal:
python "main GUI.py"


# ========================================
# VERIFICACIÓN DE DEPENDENCIAS
# ========================================
pip list
# Busca estas librerías en la lista:
# - openai-whisper
# - PyQt5
# - pyaudio
# - numpy
# - torch
# - sounddevice


# ========================================
# CREAR requirements.txt (OPCIONAL)
# ========================================
# En el PC donde ya funciona:
pip freeze > requirements.txt

# En el nuevo PC:
pip install -r requirements.txt


# ========================================
# SI HAY ERRORES
# ========================================

# Error: "No module named 'PyQt5'"
pip install PyQt5

# Error: "No module named 'pyaudio'"
pip install pipwin
pipwin install pyaudio

# Error: "ffmpeg not found"
# 1. Reinstala ffmpeg
# 2. REINICIA el terminal/IDE
# 3. Verifica: ffmpeg -version

# Error: "No module named 'whisper'"
pip install openai-whisper


# ========================================
# NOTAS IMPORTANTES
# ========================================

# 1. El modelo Whisper (~1.5GB) se descarga AUTOMÁTICAMENTE
#    la primera vez que ejecutes test_whisper_instalacion.py
#    o uses la grabación en la aplicación

# 2. El modelo se guarda en:
#    C:\Users\TuUsuario\.cache\whisper\

# 3. La segunda vez que uses la aplicación, cargará mucho más rápido
#    porque el modelo ya está en cache

# 4. RECUERDA reiniciar el terminal después de instalar ffmpeg

# 5. Si la transcripción es muy lenta, cambia el modelo a "small"
#    en los archivos:
#    - gui/pantalla_preguntas.py (línea ~238)
#    - gui/ventana_detalle_edit_pregunta_interno.py (línea ~268)
